# Стартовый скрипт для тестирования VFS эмулятора
# Демонстрация всех команд этапа 4

# 1. Информация о VFS
vfs-info

# 2. Список файлов в корневой директории
ls /

# 3. Смена директории и просмотр содержимого
cd documents
ls

# 4. Возврат в корень и просмотр другого файла
cd /
cat /readme.txt

# 5. Просмотр начала файла с помощью head (первые 5 строк)
head -n 5 /documents/file1.txt

# 6. Просмотр начала файла с помощью head (по умолчанию 10 строк)
head /documents/file2.txt

# 7. Попытка перехода в несуществующую директорию
cd nonexistent_directory

# 8. Попытка чтения несуществующего файла
cat /nonexistent_file.txt

# 9. Попытка head для несуществующего файла
head -n 3 /nonexistent_file.txt

# 10. Использование относительных путей
cd documents
ls
cat file1.txt
head -n 2 file2.txt

# 11. Возврат в корневую директорию и финальный список
cd /
ls

# 12. Финальная информация о VFS
vfs-info

# Завершение работы
exit
